   4.2、查询：
        SearchResult searchResult = jestService.search(jestClient, indexName, get<T>Query(T t));
        List<SearchResult.Hit<T, Void>> hits = searchResult.getHits(T.class);
        List<T> objs = new ArrayList<>();
            hits.forEach(o -> {
            objs.add(o.source);
        });
   4.3、添加方法：
        public static String get<T>Query(T t) {
            SearchSourceBuilder searchSourceBuilder = new SearchSourceBuilder();
            QueryBuilder queryBuilder = QueryBuilders.boolQuery();
            if (StrUtil.isNotEmpty(t.getAttribute())) {
                ((BoolQueryBuilder) queryBuilder).must(QueryBuilders.boolQuery().should(QueryBuilders.termQuery("attribute", t.getAttribute())));
            }
            searchSourceBuilder.query(queryBuilder);
            searchSourceBuilder.from((t.getPageNum() -1 ) * t.getPageSize());
            searchSourceBuilder.size(t.getPageSize());
            return searchSourceBuilder.toString();
        }
